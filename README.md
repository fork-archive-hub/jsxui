Primitive elements to build isomorphic user interfaces in React.

## Usage

```jsx
import React from 'react'
import { Stack, Text } from 'jsx-ui'

export default () => {
  return (
    <Stack>
      <Text color="orange">JSX UI</Text>
    </Stack>
  )
}
```
